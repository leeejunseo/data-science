```python
    features[1] = np.sqrt(x[-1]**2 + y[-1]**2) / 1000  # final_range_km
    features[2] = np.abs(np.rad2deg(gamma[-1]))  # impact_angle_deg
    features[3] = t[-1]  # total_flight_time
    
    # ì†ë„/ë§ˆí•˜
    features[4] = np.max(V)  # max_velocity
    features[5] = V[-1]  # terminal_velocity
    features[6] = np.max(mach)  # max_mach
    features[7] = (np.max(V) - V[-1]) / (np.max(V) + 1e-6)  # velocity_loss_ratio
    
    # ê°ì†/ê¸°ë™/íš¨ìœ¨
    features[8] = np.min(np.gradient(V, t))  # max_deceleration
    features[9] = np.mean(curvature)  # ground_track_curvature
    features[10] = final_range / path_length  # path_efficiency
    features[11] = KE / PE  # energy_ratio
    
    # 6DOF ê¸°ë™ì„±
    features[12] = np.std(np.rad2deg(alpha))  # alpha_std_deg
    features[13] = np.max(np.abs(np.rad2deg(q)))  # q_max_deg_s
    features[14] = np.corrcoef(alpha, q)[0,1]  # alpha_q_correlation
    
    return features
```
